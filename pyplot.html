# Generate a sequence of 5 random integers
rand_sequence = [randint(1, 100) for _ in range(5)]
print("Random sequence:", rand_sequence)

# Create a scatter plot of the random sequence
plt.scatter(range(len(rand_sequence)), rand_sequence, color='blue', marker='o')
plt.title('Random Integer Scatter Plot')
plt.xlabel('Index')
plt.ylabel('Random Integer Value')
plt.show()

# Create a line plot of the random sequence
plt.plot(rand_sequence, color='green', marker='o')
plt.title('Random Integer Line Plot')
plt.xlabel('Index')
plt.ylabel('Random Integer Value')
plt.grid(True)
plt.show()
